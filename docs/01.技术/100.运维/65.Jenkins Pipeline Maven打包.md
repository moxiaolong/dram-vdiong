## Pipeline
```groovy
env.serviceDir = "."

stage('maven package') {
    dir(env.serviceDir) {
        //安装maven
        withMaven(maven: 'maven') {
            env.PATH = "/root/tools/hudson.tasks.Maven_MavenInstallation/maven/bin:${env.PATH}"
            //maven打包
            sh "mvn package -Dmaven.test.skip=true"
            //删除源码包
            sh "rm -f ./target/*-sources.jar"
        }
    }
}

```
